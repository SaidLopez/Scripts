Sub Toogle_Quest_Items
exevent popup #charid
wait 10
set %Contsize #contsize
str mid %contsize 5 3
wait 10
set %Tooglex #contposx + 50
set %Toogley #contposy + #strres - 20
click %Tooglex %Toogley
wait 10
Return

Sub Target_NPC
Target_NPC:
Finditem %NPCQUESTER G_18
if #findkind <> -1
{
set #lobjectid %NPCQUESTER
event macro 17 0
wait 15
}
Return

sub Toogle_Quest

call kalocr.txt getMLQInfo QuestOffer #contposx #contposy

if mother in %1
{
 set %clickx #contposx + 130
 set %clicky #contposy + 400
 click %clickx %clicky
 }
else
{
goto start
}
Return

sub target_tinker
finditem GTL C_ , #backpackid
  for #findindex 1 #findcnt
  {
    wait 8
    set #ltargetid #findid
    set #LTargetKind 1
    target 5s
    wait 8
    event macro 22 0
  }
  wait 10
  finditem ZVH C_ , #backpackid
  for #findindex 1 #findcnt
  {
    if #findcol = 0
    {
      wait 8
      set #ltargetid #findid
      set #LTargetKind 1
      target 5s
      wait 8
      event macro 22 0
    }
  }
  key esc
return

sub Get_Reward
Accept_NPC:
Finditem %NPCQUESTER G_18
if #findkind <> -1
{
if #finddist > 18
{
;move #findx #findy 0 1s
goto Accept_Npc
}
set #lobjectid %NPCQUESTER
event macro 17 0
wait 15
}
wait 10
set %clickx #contposx + 130
set %clicky #contposy + 400
click %clickx %clicky
wait 10
set %clickx #contposx + 130
set %clicky #contposy + 400
click %clickx %clicky
wait 10
key esc
wait 10
Return

sub Sort_Rewards
Sort_Rewards:
menu get  EnableRegRecipes
if #menures = #true
{
set %recipesuccess true
}
if #menures = #false
{
set %recipesuccess false
}
Finditem ZJF C_ , #backpackid
if #findkind <> -1
{
set %Currentbag #findid
wait 6

set #lobjectid %Currentbag
event macro 17 0
wait 15

Finditem %talisman C_ , %Currentbag
if #findkind <> -1
{
set %Findid #findid
exevent drag %Findid 1
wait 1s
exevent dropc %RareRecipebag
wait 1s
}
wait 15
finditem %runic C_ , %Currentbag
if #findkind <> -1
{
set %findid #findid
event property %findid
if Pendant in #property
{
exevent drag %findid 1
wait 5
exevent dropc %RareRecipebag
wait 5
}
if Compendium in #property
{
exevent drag %findid 1
wait 5
exevent dropc %RareRecipebag
wait 5
}
}
exevent drag %Currentbag 1
wait 1s
exevent dropc %Trashbag
wait 1s
Return

sub Create_tinker
while #findcnt <= 11
{
finditem GTL C_ , #backpackid
set #lobjectid #findid
event macro 17 0
wait 5
contpos 0 0
click 285 450
wait 1s
finditem GTL C_ , #backpackid
}
contpos 0 0
wait 5
click 25 450
return
