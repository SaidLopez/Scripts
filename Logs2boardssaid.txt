set %wood_col 0
set %oak_col 2010
set %ash_col 1191
set %yew_col 1192
set %heart_col 1193
set %blood_col 1194
set %frost_col 1151

finditem BSF_ZRF_UOH C_ , #CHARID
if #FINDKIND = -1
{
  display ok Can't find your axe, put the spell channeling axe you will be using in your paperdoll.
  stop
}
set %axe #FINDID

chooseskill CARP
set !skill #SKILL
chooseskill LUMB
if #SKILL > !skill
  set !skill #SKILL

Createboards_loop1:
  finditem ZLK C ; look through logs in main pack.
  if #FINDCNT > 0
  {
    set !convert_it #FALSE
    set !log_color #FINDCOL
    set !log_id #FINDID
    set %log_stack 150
    set !log_bag #FINDBAGID

    if !log_color = %wood_col
      set !convert_it #TRUE
    if !log_color = %oak_col && !skill >= 650
      set !convert_it #TRUE
    if !log_color = %ash_col && !skill >= 800
      set !convert_it #TRUE
    if !log_color = %yew_col && !skill >= 950
      set !convert_it #TRUE
    if !log_color = %heart_col && !skill >= 1000
      set !convert_it #TRUE
    if !log_color = %blood_col && !skill >= 1000
      set !convert_it #TRUE
    if !log_color = %frost_col && !skill >= 1000
      set !convert_it #TRUE

    if !convert_it = #TRUE
    {
      exevent drag !log_id %log_stack
      wait 10
      event macro 22 0
      wait 10
      exevent dropc #BACKPACKID
      wait 10
      set #LTARGETID #FINDID
      set #LOBJECTID %axe
      set #LTARGETKIND 1
      event macro 17 0
      target 5s
      event macro 22 0
      wait 25
      finditem TLK C_ , #BACKPACKID
      if #FINDCNT > 0
      {
        for #FINDINDEX 1 #FINDCNT
        {
          if #FINDCOL = !log_color
          {
            exevent drag #FINDID #FINDSTACK
            wait 10
            exevent dropc !log_bag
            wait 20
            break
          }
        }
      }
    }
    ignoreitem !log_id strangewood
    wait 5
    goto Createboards_loop1
  }
stop
