;########### Checks ###########

;########### Sacarbod ###########
sub Sacarbod
  finditem %b_bod C_ , #backpackid
  if #findkind <> -1
    return
  set #lobjectid %ibook
  event macro 17 0
  gosub waitforgump 615_454
  set %my_x #contposx + 40
  set %my_y #contposy + 101
  click %my_x %my_y
  gosub waitforgump 615_454
  click %my_x %my_y r
return

;########### Guardarbod ###########
sub Guardarbod
  finditem %b_bod C_ , #backpackid
  if #findkind = -1
    return
  for #FINDINDEX 1 #FINDCNT
  {
    Exevent Drag #findid 1
    wait 10
    Exevent Dropc %end_book
    wait 20
  }
  gosub waitforgump 615_454
  wait 10
  set %my_x_1 #contposx + 50
  set %my_y_1 #contposy + 50
  click %my_x_1 %my_y_1 r
return

;########### waitforgump ###########
;usage gosub waitforgump #contsize
sub waitforgump
  set #result 0
  set %Csize %1
  set %timeout #SCNT + 7
  repeat
    until #CONTSIZE = %Csize || %timeout =< #SCNT
    if %timeout = #SCNT
      return no
return
